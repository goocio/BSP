\begin{thm}[Dominated Convergence Theorem] \label{thm:DCT}
    Let \((f_n)\) be a sequence of measurable functions such that:
    \begin{enumerate}
        \item \((f_n(x))\) converges a.e.~to a limit \(f(x)\),
        \item there is an integrable function \(g\) such that, for each \(n\), \(\abs{f_n (x)} \leqslant g(x)~\mathrm{a.e.}\).
    \end{enumerate}
    Then \(f\) is integrable, and \(\int f = \lim_{n\to\infty} \int f_n\).
\end{thm}
\flushproof\begin{proof}
    The proof is trivial! Just biject it to a skew-symmetric residue class whose elements are twice-differentiable algebras.
\end{proof}
